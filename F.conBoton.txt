using UnityEngine;
using UnityEngine.SceneManagement;
using UnityEngine.UI;

public class IniciarScript : MonoBehaviour
{
    public Button iniciarButton;

    void Start()
    {
        // Esto solo se ejecutará si arrastras un botón al hueco del Inspector
        if (iniciarButton != null)
        {
            iniciarButton.onClick.AddListener(empezar);
        }
    }

    void empezar()
    {
        SceneManager.LoadScene(1);
    }

    // --- NUEVA FUNCIÓN PARA EL BOTÓN DEL FINAL ---
    public void irAlFinal()
    {
        Time.timeScale = 1f; // Muy importante para que el juego no siga pausado
        SceneManager.LoadScene("final"); // Asegúrate de que se llame así en Build Settings
    }
}
Como quieres que el botón aparezca solo cuando el tiempo se acabe, sigue este orden en el editor de Unity:

Crea el Panel: Haz clic derecho en tu Canvas -> UI -> Panel. Ponle de nombre PanelGameOver.

Crea el Botón: Dentro de ese panel, crea un botón que diga "Ir al Final".

Desactiva el Panel: Selecciona el PanelGameOver y desmarca la casilla que hay al lado de su nombre en el Inspector. Se volverá invisible.
el tiempo:using TMPro;
using UnityEngine;
using UnityEngine.UI;

public class tiempoScript : MonoBehaviour
{
    [Header("Configuración de Tiempo")]
    public float tiempo = 60f;
    public Image relog;
    public TextMeshProUGUI puntuacion;

    [Header("Interfaz de Fin de Juego")]
    // Arrastra aquí el Panel desde la jerarquía de Unity
    public GameObject panelGameOver; 

    void Update()
    {
        // Si aún queda tiempo, el juego sigue normal
        if (tiempo > 0f)
        {
            tiempo -= Time.deltaTime;
            
            // Actualizamos el texto de puntuación
            puntuacion.text = (6000 - (int)(tiempo * 100)).ToString();
            
            // Actualizamos la imagen del reloj (fillAmount va de 0 a 1)
            relog.fillAmount = tiempo / 60f;
        }
        else 
        {
            // CUANDO EL TIEMPO LLEGA A CERO:
            FinalizarPartida();
        }
    }

    void FinalizarPartida()
    {
        // 1. Detenemos el movimiento de todo el juego
        Time.timeScale = 0f; 

        // 2. Activamos el panel que contiene el botón
        if (panelGameOver != null)
        {
            panelGameOver.SetActive(true);
        }
        
        // Opcional: Aseguramos que el tiempo no baje de 0 en el texto
        tiempo = 0;
    }
}
Pasos finales en el Editor (Para que no falle):
En el Inspector del script de Tiempo: Verás el nuevo hueco Panel Game Over. Arrastra ahí el Panel que creaste (el que tiene el botón dentro).

Estado inicial: Asegúrate de que ese Panel esté desactivado (el tick de arriba a la izquierda en el Inspector del Panel) antes de darle a Play.

El Botón: Al botón que está dentro del panel, asígnale la función irAlFinal() de tu otro script IniciarScript, tal como lo tenías planeado.
